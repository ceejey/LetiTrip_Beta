/**
         * Polar H6 Connection
         */
        final PolarHandler polar = new PolarHandler(this, (TextView) findViewById(R.id.txtHeartRate2), (TextView) findViewById(R.id.txtStatus));
        Button btnSearch = (Button)findViewById(R.id.btnSearchPolarH6);
        Button btnConnect = (Button)findViewById(R.id.btnConnectPolar);
        Button btnDisconnect = (Button)findViewById(R.id.btnDisconnectPolarH6);
        Button btnToggleHeartRate = (Button)findViewById(R.id.btnToggleHeartRate);
        btnSearch.setOnClickListener(
                new Button.OnClickListener() {
                    public void onClick(View v) {
                            polar.searchPolarDevice();
                    }
                }
        );
        btnConnect.setOnClickListener(
                new Button.OnClickListener() {
                    public void onClick(View v) {
                            polar.connectToPolarDevice();
                    }
                }
        );
        btnDisconnect.setOnClickListener(
                new Button.OnClickListener() {
                    public void onClick(View v) {
                        polar.disconnectFromPolarDevice();
                    }
                }
        );
        btnToggleHeartRate.setOnClickListener(
                new Button.OnClickListener() {
                    public void onClick(View v) {
                        if(!polar.isReceiveHeartRate())
                            polar.receiveHeartRate();
                        else
                            polar.stopReceivingHeartRate();
                    }
                }
        );
